function calculateCoverage(planType, baseAmount) {
  if (planType === "standard") {
    return baseAmount;
  } else if (planType === "premium") {
    return baseAmount * 1.5;
  } else if (planType === "vip") {
    return baseAmount * 2;
  }
  return "Invalid plan type";
}
